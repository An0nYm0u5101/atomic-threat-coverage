    selection1:
        Image: '*\rundll32.exe'

### powershell
    
```
Get-WinEvent | where {(($_.message -match "Image.*.*\\\\rundll32.exe" -and $_.message -match "CommandLine.*.*,dll_u") -or $_.message -match "CommandLine.*.* -export dll_u .*") } | select TimeCreated,Id,RecordId,ProcessId,MachineName,Message
```
